# Multi-site Nginx configuration for a single VPS.
#
# Deploy target:
#   /etc/nginx/sites-available/default
#
# Host routing:
#   server.key-ching.com  -> 127.0.0.1:2999  (SSL already issued)
#   server.scramblurr.com -> 127.0.0.1:3000 and /py -> 127.0.0.1:5000
#   server.drauwper.com   -> 127.0.0.1:4000
#   server.iotachat.app   -> 127.0.0.1:8000  (SSL via certbot - see Phase 2 below)
#
# *** server.iota-chat.com has been retired (replaced by server.iotachat.app) ***

# --- HTTP -> HTTPS redirect (domains with active certs) ---
server {
    server_name server.key-ching.com server.scramblurr.com server.drauwper.com server.iotachat.app server.faceblurr.com server.hostagechess.boredgamez.org server.prolifer8.app;
    return 301 https://$host$request_uri;

    listen [::]:443 ssl ipv6only=on; # managed by Certbot
    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/server.prolifer8.app/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/server.prolifer8.app/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name server.key-ching.com;

    ssl_certificate /etc/letsencrypt/live/server.key-ching.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/server.key-ching.com/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    client_max_body_size 500M;
    client_body_timeout 300s;
    proxy_connect_timeout 300s;
    proxy_send_timeout 300s;
    proxy_read_timeout 300s;
    send_timeout 300s;

    location / {
        proxy_pass http://127.0.0.1:2999;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_request_buffering off;
        proxy_buffering off;
    }
}
